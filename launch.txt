       import { schnorr } from '@noble/curves/secp256k1.js';
       import * as bitcoin from 'bitcoinjs-lib';
       import * as ecc from '@bitcoinerlab/secp256k1';
       bitcoin.initEccLib(ecc);
       const priv = Buffer.from('<PRIVKEY_HEX>', 'hex');
       const xpub = Buffer.from(schnorr.getPublicKey(priv));
       const { address } = bitcoin.payments.p2tr({ internalPubkey: xpub, network: bitcoin.networks.bitcoin });
       console.log(address);
